// Call different presets based on events
document.addEventListener('keydown', (event) => {
    switch(event.key) {
        case '1':
            presetManager.applyPreset('rave_mode');
            break;
        case '2':
            presetManager.applyPreset('chill_mode');
            break;
        case '3':
            presetManager.applyPreset('chaos_mode');
            break;
    }
});
